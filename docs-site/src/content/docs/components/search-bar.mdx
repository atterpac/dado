import ComponentPage from '../../../components/ComponentPage.astro';
import data from '../../api/search-bar.json';
import { getAvailableThemes } from '../../../lib/themes';

export const themes = await getAvailableThemes();

## Key behavior

`Enter` fires `onSubmit` with the current text. If the dropdown is open, `Enter` selects the highlighted result instead. `Esc` closes the dropdown; a second `Esc` clears the input.
